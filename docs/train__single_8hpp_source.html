194template<typename Value_, typename Index_, typename Label_, typename Float_, class Matrix_>
195TrainedSingle<Index_, Float_> train_single(
196 const tatami::Matrix<Value_, Index_>& ref,
197 const Label_* labels,
198 Markers<Index_> markers,
199 const TrainSingleOptions<Index_, Float_, Matrix_>& options)
200{
201 auto subset = internal::subset_to_markers(markers, options.top);
202 auto subref = internal::build_references(ref, labels, subset, options);
203 Index_ test_nrow = ref.nrow(); // remember, test and ref are assumed to have the same features.
204 return TrainedSingle<Index_, Float_>(test_nrow, std::move(markers), std::move(subset), std::move(subref));
