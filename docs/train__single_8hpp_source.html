197template<typename Value_, typename Index_, typename Label_, typename Float_, class Matrix_>
198TrainedSingle<Index_, Float_> train_single(
199 const tatami::Matrix<Value_, Index_>& ref,
200 const Label_* labels,
201 Markers<Index_> markers,
202 const TrainSingleOptions<Index_, Float_, Matrix_>& options)
203{
204 auto subset = internal::subset_to_markers(markers, options.top);
205 auto subref = internal::build_references(ref, labels, subset, options);
206 Index_ test_nrow = ref.nrow(); // remember, test and ref are assumed to have the same features.
