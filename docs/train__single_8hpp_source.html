158template<typename Float_ = double, typename Value_, typename Index_, typename Label_>
159TrainedSingle<Index_, Float_> train_single(

161 const Label_* labels,
162 Markers<Index_> markers,
163 const TrainSingleOptions& options)
164{
165 auto subset = internal::subset_to_markers(markers, options.top);
166 auto subref = internal::build_indices<Float_>(ref, labels, subset, options.num_threads);
167 Index_ test_nrow = ref.nrow(); // remember, test and ref are assumed to have the same features.
168 return TrainedSingle<Index_, Float_>(test_nrow, std::move(markers), std::move(subset), std::move(subref));
