450template<typename Value_, typename Index_, typename Label_>
451TrainedIntegrated<Index_> train_integrated(const std::vector<TrainIntegratedInput<Value_, Index_, Label_> >& inputs, const TrainIntegratedOptions& options) {

453 // Checking that the number of genes in the test dataset are consistent.
454 Index_ test_nrow = 0;
455 if (inputs.size()) {
456 test_nrow = inputs.front().test_nrow;
457 for (const auto& in : inputs) {
458 if (!sanisizer::is_equal(in.test_nrow, test_nrow)) {
459 throw std::runtime_error("inconsistent number of rows in the test dataset across entries of 'inputs'");
460 }
461 }
462 }
463
464 // For references with intersections, we need to map the marker indices to the test rows, for comparability across references.
465 const auto nrefs = inputs.size();
466 std::vector<std::vector<Index_> > remap_intersection_to_test_index;
467 for (I<decltype(nrefs)> r = 0; r < nrefs; ++r) {
468 if (inputs[r].intersection.has_value()) {
469 sanisizer::resize(remap_intersection_to_test_index, nrefs);
470 break;
471 }
472 }
473
474 // Identify the union of all marker genes as the universe, but excluding those markers that are not present in all intersections.
475 // Specifically, 'universe' contains sorted and unique row indices for the test matrix, where 'remap_test_to_universe[universe[i]] == i'.
476 // 'remap_test_to_universe[k]' defaults to the max number of rows in the test if 'k' is not present in all intersections.
477 std::vector<Index_> universe;
478 auto remap_test_to_universe = sanisizer::create<std::vector<Index_> >(test_nrow, test_nrow);
479 {
480 auto present = sanisizer::create<std::vector<char> >(test_nrow);
481 auto count_refs = sanisizer::create<std::vector<I<decltype(nrefs)> > >(test_nrow);
482 universe.reserve(test_nrow);
483
484 for (I<decltype(nrefs)> r = 0; r < nrefs; ++r) {
485 const auto& markers = inputs[r].markers;
486 const auto& inter = inputs[r].intersection;
487
488 if (inter.has_value()) {
489 auto& cur_test_remap = remap_intersection_to_test_index[r];
490 sanisizer::resize(cur_test_remap, inputs[r].ref->nrow(), test_nrow); // again, default to the max number of rows if not present in the current intersection.
491 for (const auto& pp : *inter) {
492 cur_test_remap[pp.second] = pp.first;
493 count_refs[pp.first] += 1;
494 }
495
496 for (const auto& labmrk : markers) {
497 for (const auto y : labmrk) {
498 const auto ty = cur_test_remap[y];
499 if (ty != test_nrow && !present[ty]) {
500 present[ty] = true;
501 universe.push_back(ty);
502 }
503 }
504 }
505
506 } else {
507 for (const auto& labmrk : markers) {
508 for (const auto y : labmrk) {
509 if (!present[y]) {
510 present[y] = true;
511 universe.push_back(y);
512 }
513 }
514 }
515
516 for (auto& x : count_refs) {
517 x += 1;
518 }
519 }
520 }
521
522 std::sort(universe.begin(), universe.end());
523 const auto num_universe = universe.size();
524 I<decltype(num_universe)> keep = 0;
525 for (I<decltype(num_universe)> u = 0; u < num_universe; ++u) {
526 const auto marker = universe[u];
527 if (count_refs[marker] == nrefs) {
528 universe[keep] = marker;
529 remap_test_to_universe[marker] = keep;
530 ++keep;
531 }
532 }
533 universe.resize(keep);
534 universe.shrink_to_fit();
535 }
536
537 // Remapping the per-label markers for this reference to refer to our universe.
538 auto references = sanisizer::create<std::vector<IntegratedReference<Index_> > >(nrefs);
539 for (I<decltype(nrefs)> r = 0; r < nrefs; ++r) {
540 const auto& curinput = inputs[r];
541 const auto& currefmarkers = curinput.markers;
542 const auto nlabels = currefmarkers.size();
543 auto& currefout = references[r];
544
545 const bool is_sparse = curinput.ref->is_sparse();
546 if (is_sparse) {
547 currefout.sparse.emplace(sanisizer::as_size_type<I<decltype(*(currefout.sparse))> >(nlabels));
548 } else {
549 currefout.dense.emplace(sanisizer::as_size_type<I<decltype(*(currefout.dense))> >(nlabels));
550 }
551
552 auto get_markers = [&](I<decltype(nlabels)> l) -> std::vector<Index_>& {
553 if (is_sparse) {
554 return (*(currefout.sparse))[l].markers;
555 } else {
556 return (*(currefout.dense))[l].markers;
557 }
558 };
559
560 if (curinput.intersection.has_value()) {
561 auto& cur_test_remap = remap_intersection_to_test_index[r];
562 for (I<decltype(nlabels)> l = 0; l < nlabels; ++l) {
563 const auto& curlabmarkers = currefmarkers[l];
564 auto& markers = get_markers(l);
565 markers.reserve(curlabmarkers.size());
566 for (const auto y : curlabmarkers) {
567 const auto ty = cur_test_remap[y];
568 if (ty != test_nrow) { // ignoring marker that can't be mapped via the current intersection.
569 const auto universe_index = remap_test_to_universe[ty];
570 if (universe_index != test_nrow) { // ignoring genes not present in all intersections.
571 markers.push_back(universe_index);
572 }
573 }
574 }
575 }
576
577 } else {
578 for (I<decltype(nlabels)> l = 0; l < nlabels; ++l) {
579 const auto& curlabmarkers = currefmarkers[l];
580 auto& markers = get_markers(l);
581 markers.reserve(curlabmarkers.size());
582 for (const auto y : curlabmarkers) {
583 const auto universe_index = remap_test_to_universe[y];
584 if (universe_index != test_nrow) { // ignoring genes not present in all intersections.
585 markers.push_back(universe_index);
586 }
587 }
588 }
589 }
590 }
591
592 // Not needed after this so we try to free its allocation for recycling in the next malloc call.
593 // Specifically, I want to give a chance for this memory to be re-used in train_integrated_per_reference_intersect.
594 remap_intersection_to_test_index.clear();
595
596 // Now, we create ranked vectors for each profile in the reference.
597 for (I<decltype(nrefs)> r = 0; r < nrefs; ++r) {
598 const auto& curinput = inputs[r];
599 const auto nlabels = curinput.markers.size();
600 auto& currefout = references[r];
601
602 std::vector<Index_> positions;
603 const Index_ NC = curinput.ref->ncol();
604 sanisizer::reserve(positions, NC);
605 auto samples_per_label = sanisizer::create<std::vector<Index_> >(nlabels);
606 for (Index_ c = 0; c < NC; ++c) {
607 auto& pos = samples_per_label[curinput.labels[c]];
608 positions.push_back(pos);
609 ++pos;
610 }
611
612 if (curinput.ref->is_sparse()) {
613 auto negative_ranked = sanisizer::create<std::vector<std::vector<RankedVector<Index_, Index_> > > >(nlabels);
614 auto positive_ranked = sanisizer::create<std::vector<std::vector<RankedVector<Index_, Index_> > > >(nlabels);
615 for (I<decltype(nlabels)> l = 0; l < nlabels; ++l) {
616 const auto num_samples = samples_per_label[l];
617 sanisizer::resize(negative_ranked[l], num_samples);
618 sanisizer::resize(positive_ranked[l], num_samples);
619 }
620
621 if (curinput.intersection) {
622 train_integrated_per_reference_intersect<true>(curinput, remap_test_to_universe, test_nrow, options, positions, negative_ranked, positive_ranked);
623 } else {
624 train_integrated_per_reference_simple<true, Value_>(curinput, universe, remap_test_to_universe, options, positions, negative_ranked, positive_ranked);
625 }
626
627 for (I<decltype(nlabels)> l = 0; l < nlabels; ++l) {
628 auto& curlabout = (*(currefout.sparse))[l];
629 const auto num_samples = samples_per_label[l];
630 curlabout.num_samples = num_samples;
631
632 I<decltype(curlabout.negative_ranked.size())> num_neg = 0;
633 for (const auto& x : negative_ranked[l]) {
634 num_neg = sanisizer::sum<I<decltype(num_neg)> >(num_neg, x.size());
635 }
636
637 I<decltype(curlabout.positive_ranked.size())> num_pos = 0;
638 for (const auto& x : positive_ranked[l]) {
639 num_pos = sanisizer::sum<I<decltype(num_pos)> >(num_pos, x.size());
640 }
641
642 curlabout.negative_ranked.reserve(num_neg);
643 curlabout.negative_indptrs.reserve(sanisizer::sum<I<decltype(curlabout.negative_indptrs.size())> >(num_samples, 1));
644 curlabout.negative_indptrs.push_back(0);
645 for (const auto& x : negative_ranked[l]) {
646 curlabout.negative_ranked.insert(curlabout.negative_ranked.end(), x.begin(), x.end());
647 curlabout.negative_indptrs.push_back(curlabout.negative_ranked.size());
648 }
649
650 curlabout.positive_ranked.reserve(num_pos);
651 curlabout.positive_indptrs.reserve(sanisizer::sum<I<decltype(curlabout.positive_indptrs.size())> >(num_samples, 1));
652 curlabout.positive_indptrs.push_back(0);
653 for (const auto& x : positive_ranked[l]) {
654 curlabout.positive_ranked.insert(curlabout.positive_ranked.end(), x.begin(), x.end());
655 curlabout.positive_indptrs.push_back(curlabout.positive_ranked.size());
656 }
657 }
658
659 } else {
660 auto out_ranked = sanisizer::create<std::vector<std::vector<RankedVector<Index_, Index_> > > >(nlabels);
661 for (I<decltype(nlabels)> l = 0; l < nlabels; ++l) {
662 const auto num_samples = samples_per_label[l];
663 sanisizer::resize(out_ranked[l], num_samples);
664 }
665
666 if (curinput.intersection) {
667 train_integrated_per_reference_intersect<false>(curinput, remap_test_to_universe, test_nrow, options, positions, out_ranked, true);
668 } else {
669 train_integrated_per_reference_simple<false, Value_>(curinput, universe, remap_test_to_universe, options, positions, out_ranked, true);
670 }
671
672 for (I<decltype(nlabels)> l = 0; l < nlabels; ++l) {
673 auto& curlabout = (*(currefout.dense))[l];
674 curlabout.num_samples = samples_per_label[l];
675 curlabout.all_ranked.reserve(sanisizer::product<I<decltype(curlabout.all_ranked.size())> >(universe.size(), curlabout.num_samples));
676 for (const auto& x : out_ranked[l]) {
677 curlabout.all_ranked.insert(curlabout.all_ranked.end(), x.begin(), x.end());
678 }
679 }
680 }
681 }
682
683 return TrainedIntegrated<Index_>(test_nrow, std::move(universe), std::move(references));
684}
