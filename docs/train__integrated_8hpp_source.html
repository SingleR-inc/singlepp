265template<bool ref_sparse_, typename Value_, typename Index_, typename Label_>
266void train_integrated_per_reference_simple(
267 const TrainIntegratedInput<Value_, Label_, Index_>& input,
268 const std::vector<Index_>& universe,
269 const std::vector<Index_>& remap_test_to_universe,
270 const TrainIntegratedOptions& options,
271 const std::vector<Index_>& positions,
272 std::vector<std::vector<RankedVector<Index_, Index_> > >& out_ranked,
273 typename std::conditional<ref_sparse_, std::vector<std::vector<RankedVector<Index_, Index_> > >&, bool>::type other_ranked
274) {
275 const auto& ref = *(input.ref);
276 const auto NC = ref.ncol();
277 const auto num_universe = universe.size();
278
279 // 'universe' technically refers to the row indices of the test matrix,
280 // but in simple mode, the rows of the test and reference are the same, so we can use it directly here.
281 tatami::VectorPtr<Index_> universe_ptr(tatami::VectorPtr<Index_>{}, &universe);
282
283 tatami::parallelize([&](int, Index_ start, Index_ len) {
284 auto vbuffer = sanisizer::create<std::vector<Value_> >(num_universe);
285 auto ibuffer = [&](){
286 if constexpr(ref_sparse_) {
287 return sanisizer::create<std::vector<Index_> >(num_universe);
288 } else {
289 return false;
290 }
291 }();
292
293 RankedVector<Value_, Index_> tmp_ranked;
294 tmp_ranked.reserve(num_universe);
295
296 auto ext = tatami::consecutive_extractor<ref_sparse_>(ref, false, start, len, universe_ptr);
297 for (Index_ c = start, end = start + len; c < end; ++c) {
298 tmp_ranked.clear();
299
300 if constexpr(ref_sparse_) {
301 auto info = ext->fetch(vbuffer.data(), ibuffer.data());
302 for (I<decltype(info.number)> i = 0; i < info.number; ++i) {
303 const auto remapped = remap_test_to_universe[info.index[i]];
304 assert(sanisizer::is_less_than(remapped, num_universe));
305 tmp_ranked.emplace_back(info.value[i], remapped);
306 }
307 } else {
308 auto ptr = ext->fetch(vbuffer.data());
309 for (I<decltype(num_universe)> i = 0; i < num_universe; ++i) {
310 tmp_ranked.emplace_back(ptr[i], i); // a.k.a. remap_test_to_universe[universe[i]].
311 }
312 }
313
314 std::sort(tmp_ranked.begin(), tmp_ranked.end());
315
316 if constexpr(ref_sparse_) {
317 const auto tStart = tmp_ranked.begin(), tEnd = tmp_ranked.end();
318 auto zero_ranges = find_zero_ranges<Value_, Index_>(tStart, tEnd);
319 simplify_ranks<Value_, Index_>(tStart, zero_ranges.first, out_ranked[input.labels[c]][positions[c]]);
320 simplify_ranks<Value_, Index_>(zero_ranges.second, tEnd, other_ranked[input.labels[c]][positions[c]]);
321 } else {
322 simplify_ranks(tmp_ranked, out_ranked[input.labels[c]][positions[c]]);
323 }
324 }
325 }, NC, options.num_threads);
326}

329void train_integrated_per_reference_intersect(
330 const TrainIntegratedInput<Value_, Label_, Index_>& input,
331 const std::vector<Index_>& remap_test_to_universe,

344 for (const auto& pair : *(input.intersection)) {
345 const auto rdex = remap_test_to_universe[pair.first];
346 if (rdex != test_nrow) {
347 ref_subset.push_back(pair.second);
348 remap_ref_subset_to_universe[pair.second] = rdex;

353 typename std::conditional<ref_sparse_, bool, std::vector<Index_> >::type remap_dense_to_universe;
354 if constexpr(!ref_sparse_) {
355 remap_dense_to_universe.reserve(ref_subset.size());
356 for (auto r : ref_subset) {
357 remap_dense_to_universe.push_back(remap_ref_subset_to_universe[r]);
358 }
359 }

361 tatami::VectorPtr<Index_> to_extract_ptr(tatami::VectorPtr<Index_>{}, &ref_subset);
362 tatami::parallelize([&](int, Index_ start, Index_ len) {

382 for (I<decltype(info.number)> i = 0; i < info.number; ++i) {
383 tmp_ranked.emplace_back(info.value[i], remap_ref_subset_to_universe[info.index[i]]);
384 }
385 } else {
386 auto ptr = ext->fetch(vbuffer.data());
387 for (I<decltype(ref_subset_size)> i = 0; i < ref_subset_size; ++i) {
388 tmp_ranked.emplace_back(ptr[i], remap_dense_to_universe[i]);

392 std::sort(tmp_ranked.begin(), tmp_ranked.end());
393
394 if constexpr(ref_sparse_) {
395 const auto tStart = tmp_ranked.begin(), tEnd = tmp_ranked.end();

419template<typename Value_, typename Index_, typename Label_>
420TrainedIntegrated<Index_> train_integrated(const std::vector<TrainIntegratedInput<Value_, Index_, Label_> >& inputs, const TrainIntegratedOptions& options) {
421 TrainedIntegrated<Index_> output;
422 const auto nrefs = inputs.size();
423 sanisizer::resize(output.references, nrefs);
424
425 // Checking that the number of genes in the test dataset are consistent.
426 output.test_nrow = 0;
427 if (inputs.size()) {
428 output.test_nrow = inputs.front().test_nrow;
429 for (const auto& in : inputs) {
430 if (!sanisizer::is_equal(in.test_nrow, output.test_nrow)) {
431 throw std::runtime_error("inconsistent number of rows in the test dataset across entries of 'inputs'");
432 }
433 }
434 }
435
436 // Identify the union of all marker genes as the universe, but excluding those markers that are not present in intersections.
437 // Note that 'output.universe' contains sorted and unique row indices for the test matrix, where 'remap_test_to_universe[universe[i]] == i'.
438 auto remap_test_to_universe = sanisizer::create<std::vector<Index_> >(output.test_nrow, output.test_nrow);
439 {
440 auto present = sanisizer::create<std::vector<char> >(output.test_nrow);
441 auto count_refs = sanisizer::create<std::vector<I<decltype(nrefs)> > >(output.test_nrow);
442 output.universe.reserve(output.test_nrow);
443
444 for (const auto& in : inputs) {
445 const auto& markers = *(in.ref_markers);
446 const auto& test_subset = *(in.test_subset);
447
448 for (const auto& labmrk : markers) {
449 for (const auto& mrk : labmrk) {
450 for (const auto y : mrk) {
451 const auto ty = test_subset[y];
452 if (!present[ty]) {
453 present[ty] = true;
454 output.universe.push_back(ty);
455 }
456 }
457 }
458 }
459
460 if (in.intersection) {
461 for (const auto& pp : *(in.intersection)) {
462 count_refs[pp.first] += 1;
463 }
464 } else {
465 for (auto& x : count_refs) {
466 x += 1;
467 }
468 }
469 }
470
471 std::sort(output.universe.begin(), output.universe.end());
472 const auto num_universe = output.universe.size();
473 I<decltype(num_universe)> keep = 0;
474 for (I<decltype(num_universe)> u = 0; u < num_universe; ++u) {
475 const auto marker = output.universe[u];
476 if (count_refs[marker] == nrefs) {
477 output.universe[keep] = marker;
478 remap_test_to_universe[marker] = keep;
479 ++keep;
480 }
481 }
482 output.universe.resize(keep);
483 output.universe.shrink_to_fit();
484 }
485
486 auto is_active = sanisizer::create<std::vector<char> >(output.test_nrow);
487 std::vector<Index_> active_genes;
488 active_genes.reserve(output.test_nrow);
489
490 for (I<decltype(nrefs)> r = 0; r < nrefs; ++r) {
491 const auto& curinput = inputs[r];
492 const auto& currefmarkers = *(curinput.ref_markers);
493 const auto& test_subset = *(curinput.test_subset);
494 const auto nlabels = currefmarkers.size();
495 auto& currefout = output.references[r];
496
497 const Index_ NC = curinput.ref->ncol();
498 const bool is_sparse = curinput.ref->is_sparse();
499 if (is_sparse) {
500 currefout.sparse.emplace(sanisizer::as_size_type<I<decltype(*(currefout.sparse))> >(nlabels));
501 } else {
502 currefout.dense.emplace(sanisizer::as_size_type<I<decltype(*(currefout.dense))> >(nlabels));
503 }
504
505 // Assembling the per-label markers for this reference.
506 for (I<decltype(nlabels)> l = 0; l < nlabels; ++l) {
507 active_genes.clear();
508 for (const auto& labmark : currefmarkers[l]) {
509 for (const auto y : labmark) {
510 const auto ty = test_subset[y];
511 if (!is_active[ty]) {
512 is_active[ty] = true;
513 active_genes.push_back(ty);
514 }
515 }
516 }
517
518 std::vector<Index_> markers;
519 markers.reserve(active_genes.size());
520
521 for (const auto a : active_genes) {
522 const auto universe_index = remap_test_to_universe[a];
523 if (universe_index != output.test_nrow) { // ignoring genes not in the intersection.
524 markers.push_back(universe_index);
525 }
526 is_active[a] = false;
527 }
528
529 if (is_sparse) {
530 (*(currefout.sparse))[l].markers.swap(markers);
531 } else {
532 (*(currefout.dense))[l].markers.swap(markers);
533 }
534 }
535
536 // Pre-allocating the ranked vectors.
537 std::vector<Index_> positions;
538 positions.reserve(NC);
539 auto samples_per_label = sanisizer::create<std::vector<Index_> >(nlabels);
540 for (Index_ c = 0; c < NC; ++c) {
541 auto& pos = samples_per_label[curinput.labels[c]];
542 positions.push_back(pos);
543 ++pos;
544 }
545
546 if (curinput.ref->is_sparse()) {
547 auto negative_ranked = sanisizer::create<std::vector<std::vector<RankedVector<Index_, Index_> > > >(nlabels);
548 auto positive_ranked = sanisizer::create<std::vector<std::vector<RankedVector<Index_, Index_> > > >(nlabels);
549 for (I<decltype(nlabels)> l = 0; l < nlabels; ++l) {
550 const auto num_samples = samples_per_label[l];
551 sanisizer::resize(negative_ranked[l], num_samples);
552 sanisizer::resize(positive_ranked[l], num_samples);
553 }
554
555 if (curinput.intersection) {
556 train_integrated_per_reference_intersect<true>(curinput, remap_test_to_universe, output.test_nrow, options, positions, negative_ranked, positive_ranked);
557 } else {
558 train_integrated_per_reference_simple<true, Value_>(curinput, output.universe, remap_test_to_universe, options, positions, negative_ranked, positive_ranked);
559 }
560
561 for (I<decltype(nlabels)> l = 0; l < nlabels; ++l) {
562 auto& curlabout = (*(currefout.sparse))[l];
563 const auto num_samples = samples_per_label[l];
564 curlabout.num_samples = num_samples;
565
566 I<decltype(curlabout.negative_ranked.size())> num_neg = 0;
567 for (const auto& x : negative_ranked[l]) {
568 num_neg = sanisizer::sum<I<decltype(num_neg)> >(num_neg, x.size());
569 }
570
571 I<decltype(curlabout.positive_ranked.size())> num_pos = 0;
572 for (const auto& x : positive_ranked[l]) {
573 num_pos = sanisizer::sum<I<decltype(num_pos)> >(num_pos, x.size());
574 }
575
576 curlabout.negative_ranked.reserve(num_neg);
577 curlabout.negative_indptrs.reserve(sanisizer::sum<I<decltype(curlabout.negative_indptrs.size())> >(num_samples, 1));
578 curlabout.negative_indptrs.push_back(0);
579 for (const auto& x : negative_ranked[l]) {
580 curlabout.negative_ranked.insert(curlabout.negative_ranked.end(), x.begin(), x.end());
581 curlabout.negative_indptrs.push_back(curlabout.negative_ranked.size());
582 }
583
584 curlabout.positive_ranked.reserve(num_pos);
585 curlabout.positive_indptrs.reserve(sanisizer::sum<I<decltype(curlabout.positive_indptrs.size())> >(num_samples, 1));
586 curlabout.positive_indptrs.push_back(0);
587 for (const auto& x : positive_ranked[l]) {
588 curlabout.positive_ranked.insert(curlabout.positive_ranked.end(), x.begin(), x.end());
589 curlabout.positive_indptrs.push_back(curlabout.positive_ranked.size());
590 }
591 }
592
593 } else {
594 auto out_ranked = sanisizer::create<std::vector<std::vector<RankedVector<Index_, Index_> > > >(nlabels);
595 for (I<decltype(nlabels)> l = 0; l < nlabels; ++l) {
596 const auto num_samples = samples_per_label[l];
597 sanisizer::resize(out_ranked[l], num_samples);
598 }
599
600 if (curinput.intersection) {
601 train_integrated_per_reference_intersect<false>(curinput, remap_test_to_universe, output.test_nrow, options, positions, out_ranked, true);
602 } else {
603 train_integrated_per_reference_simple<false, Value_>(curinput, output.universe, remap_test_to_universe, options, positions, out_ranked, true);
604 }
605
606 for (I<decltype(nlabels)> l = 0; l < nlabels; ++l) {
607 auto& curlabout = (*(currefout.dense))[l];
608 curlabout.num_samples = samples_per_label[l];
609 curlabout.all_ranked.reserve(sanisizer::product<I<decltype(curlabout.all_ranked.size())> >(output.universe.size(), curlabout.num_samples));
610 for (const auto& x : out_ranked[l]) {
611 curlabout.all_ranked.insert(curlabout.all_ranked.end(), x.begin(), x.end());
612 }
613 }
614 }
615 }
616
617 return output;
618}
