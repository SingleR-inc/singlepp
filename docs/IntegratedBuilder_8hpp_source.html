124 // Adding a reference without requiring any pruning of markers.
125 template<class Subset>
126 void add_internal_direct(const tatami::Matrix<double, int>* ref, const int* labels, const Markers& old_markers, const Subset& mat_subset) {
127 add_internal_base(ref, labels);
128
129 // Adding the markers for each label, indexed according to their
130 // position in the test matrix. This assumes that 'mat_subset' is
131 // appropriately specified to contain the test's row indices.
132 auto& new_markers = references.markers.back();
133 new_markers.reserve(old_markers.size());
134
135 for (size_t i = 0; i < old_markers.size(); ++i) {
136 const auto& cur_old_markers = old_markers[i];
137
138 std::unordered_set<int> unified;
139 for (const auto& x : cur_old_markers) {
140 unified.insert(x.begin(), x.end());
141 }
142
143 new_markers.emplace_back(unified.begin(), unified.end());
144
145 if constexpr(!std::is_same<Subset, bool>::value) {
146 auto& cur_new_markers = new_markers.back();
147 for (auto& y : cur_new_markers) {
148 y = mat_subset[y];
149 }
150 }
151 }
152 return;
153 }

155 // Adding a reference with different features from the test, requiring some
156 // pruning to remove markers that are not present in the intersection.
157 template<class Subset>
158 void add_internal_intersect(
159 const std::vector<std::pair<int, int> >& intersection,
160 const tatami::Matrix<double, int>* ref,
161 const int* labels,
162 const Markers& old_markers,
163 const Subset& ref_subset)
164 {
165 add_internal_base(ref, labels);
166
167 // Manually constructing the markers. This involves (i) pruning out the
168 // markers that aren't present in the intersection, and (ii) updating
169 // their indices so that they point to rows of 'mat', not 'ref'.
170 std::unordered_map<int, int> reverse_map;
171 for (const auto& i : intersection) {
172 reverse_map[i.second] = i.first;
173 }
174
175 auto subindex = [&](int i) -> int {
176 if constexpr(!std::is_same<Subset, bool>::value) {
177 return ref_subset[i];
178 } else {
179 return i;
180 }
181 };
182
183 auto& new_markers = references.markers.back();
184 new_markers.resize(old_markers.size());
185
186 for (size_t i = 0; i < old_markers.size(); ++i) {
187 const auto& cur_old_markers = old_markers[i];
188
189 std::unordered_set<int> unified;
190 for (const auto& x : cur_old_markers) {
191 unified.insert(x.begin(), x.end());
192 }
193
194 auto& cur_new_markers = new_markers[i];
195 cur_new_markers.reserve(unified.size());
196
197 for (auto y : unified) {
198 auto it = reverse_map.find(subindex(y));
199 if (it != reverse_map.end()) {
200 cur_new_markers.push_back(it->second);
201 }
202 }
203 }
204
205 // Constructing the mapping of mat's rows to the reference rows.
206 references.check_availability.back() = true;
207 auto& mapping = gene_mapping.back();
208 for (const auto& i : intersection) {
209 mapping[i.first] = i.second;
210 }
211 return;
212 }

413 static void fill_ranks(
414 const tatami::Matrix<double, int>* curmat,
415 const int* curlab,
416 const std::vector<int>& in_use,
417 const std::vector<int>& positions,
418 std::vector<std::vector<RankedVector<int, int> > >& cur_ranked,
419 int nthreads)
420 {
421 // If we don't need to check availability, this implies that
422 // the reference has 1:1 feature mapping to the test dataset.
423 // In that case, we can proceed quite simply.
424 tatami::parallelize([&](int, int start, int len) -> void {
425 RankedVector<double, int> tmp_ranked;
426 tmp_ranked.reserve(in_use.size());
427
428 // 'in_use' is guaranteed to be sorted and unique, see its derivation in finish().
429 // This means we can directly use it for indexed extraction.
430 auto wrk = tatami::consecutive_extractor<false>(curmat, false, start, len, in_use);
431 std::vector<double> buffer(in_use.size());
432
433 for (int c = start, end = start + len; c < end; ++c) {
434 auto ptr = wrk->fetch(c, buffer.data());
435
436 tmp_ranked.clear();
437 for (int i = 0, end = in_use.size(); i < end; ++i, ++ptr) {
438 tmp_ranked.emplace_back(*ptr, i);
439 }
440 std::sort(tmp_ranked.begin(), tmp_ranked.end());
441
442 auto& final_ranked = cur_ranked[curlab[c]][positions[c]];
443 simplify_ranks(tmp_ranked, final_ranked);
444 }
445 }, curmat->ncol(), nthreads);
446 }

448 static void fill_ranks(
449 const tatami::Matrix<double, int>* curmat,
450 const int* curlab,
451 const std::vector<int>& in_use,
452 const std::vector<int>& positions,
453 const std::unordered_map<int, int>& cur_mapping,
454 std::unordered_set<int>& cur_available,
455 std::vector<std::vector<RankedVector<int, int> > >& cur_ranked,
456 int nthreads)
457 {
458 // If we need to check availability, then we need to check
459 // the mapping of test genes to row indices of the reference.
460 std::vector<std::pair<int, int> > remapping;
461 remapping.reserve(in_use.size());
462
463 for (int i = 0, end = in_use.size(); i < end; ++i) {
464 auto it = cur_mapping.find(in_use[i]);
465 if (it != cur_mapping.end()) {
466 remapping.emplace_back(it->second, i); // using 'i' instead of 'in_use[i]', as we want to work with indices into 'in_use', not the values of 'in_use' themselves.
467 cur_available.insert(i);
468 }
469 }
470
471 std::sort(remapping.begin(), remapping.end());
472
473 // This section is just to enable indexed extraction by tatami.
474 // There's no need to consider duplicates among the
475 // 'remapping[i].first', 'cur_mapping->second' is guaranteed to be
476 // unique as a consequence of how intersect_features works.
477 std::vector<int> remapped_in_use;
478 remapped_in_use.reserve(remapping.size());
479 for (const auto& p : remapping) {
480 remapped_in_use.push_back(p.first);
481 }
482
483 tatami::parallelize([&](int, int start, int len) -> void {
484 RankedVector<double, int> tmp_ranked;
485 tmp_ranked.reserve(remapped_in_use.size());
486 std::vector<double> buffer(remapped_in_use.size());
487 auto wrk = tatami::consecutive_extractor<false>(curmat, false, start, len, remapped_in_use);
488
489 for (size_t c = start, end = start + len; c < end; ++c) {
490 auto ptr = wrk->fetch(c, buffer.data());
491
492 tmp_ranked.clear();
493 for (const auto& p : remapping) {
494 tmp_ranked.emplace_back(*ptr, p.second); // remember, 'p.second' corresponds to indices into 'in_use'.
495 ++ptr;
496 }
497 std::sort(tmp_ranked.begin(), tmp_ranked.end());
498
499 auto& final_ranked = cur_ranked[curlab[c]][positions[c]];
500 simplify_ranks(tmp_ranked, final_ranked);
501 }
502 }, curmat->ncol(), nthreads);
503 }

512 IntegratedReferences finish() {
513 // Identify the union of all marker genes.
514 auto& in_use = references.universe;
515 std::unordered_map<int, int> remap_to_universe;
516 {
517 std::unordered_set<int> in_use_tmp;
518 for (const auto& refmarkers : references.markers) {
519 for (const auto& mrk : refmarkers) {
520 in_use_tmp.insert(mrk.begin(), mrk.end());
521 }
522 }
523
524 in_use.insert(in_use.end(), in_use_tmp.begin(), in_use_tmp.end());
525 std::sort(in_use.begin(), in_use.end());
526
527 for (int i = 0, end = in_use.size(); i < end; ++i) {
528 remap_to_universe[in_use[i]] = i;
529 }
530 }
531
532 for (size_t r = 0, end = references.num_references(); r < end; ++r) {
533 auto curlab = stored_labels[r];
534 auto curmat = stored_matrices[r];
535
536 // Reindexing the markers to point to the universe.
537 auto& curmarkers = references.markers[r];
538 for (auto& outer : curmarkers) {
539 for (auto& x : outer) {
540 x = remap_to_universe.find(x)->second;
541 }
542 }
543
544 auto& cur_ranked = references.ranked[r];
545 std::vector<int> positions;
546 {
547 size_t nlabels = curmarkers.size();
548 size_t NC = curmat->ncol();
549 positions.reserve(NC);
550
551 std::vector<int> samples_per_label(nlabels);
552 for (size_t c = 0; c < NC; ++c) {
553 auto& pos = samples_per_label[curlab[c]];
554 positions.push_back(pos);
555 ++pos;
556 }
557
558 cur_ranked.resize(nlabels);
559 for (size_t l = 0; l < nlabels; ++l) {
560 cur_ranked[l].resize(samples_per_label[l]);
561 }
562 }
563
564 // Finally filling the rankings.
565 if (!references.check_availability[r]) {
566 fill_ranks(curmat, curlab, in_use, positions, references.ranked[r], nthreads);
567 } else {
568 fill_ranks(curmat, curlab, in_use, positions, gene_mapping[r], references.available[r], references.ranked[r], nthreads);
569 }
570 }
571
572 return std::move(references);
573 }
