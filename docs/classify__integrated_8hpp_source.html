115 const tatami::Matrix<Value_, Index_>& test,
116 const std::vector<const Label_*>& assigned,
117 const TrainedIntegrated<Index_>& trained,
118 ClassifyIntegratedBuffers<RefLabel_, Float_>& buffers,
119 const ClassifyIntegratedOptions<Float_>& options)
120{
121 auto NR = test.nrow();
122 auto nref = trained.num_references();
123
124 tatami::parallelize([&](size_t, Index_ start, Index_ len) -> void {
125 // We perform an indexed extraction, so all subsequent indices
126 // will refer to indices into this subset (i.e., 'trained.universe').
127 tatami::VectorPtr<Index_> universe_ptr(tatami::VectorPtr<Index_>{}, &(trained.universe));
128 auto wrk = tatami::consecutive_extractor<false>(&test, false, start, len, std::move(universe_ptr));
129 std::vector<Value_> buffer(trained.universe.size());
130
131 std::unordered_set<Index_> miniverse_tmp;
132 std::vector<Index_> miniverse;
133 internal::RankRemapper<Index_> intersect_mapping, direct_mapping;
134
135 internal::RankedVector<Value_, Index_> test_ranked_full, test_ranked;
136 test_ranked_full.reserve(NR);
137 test_ranked.reserve(NR);
138 internal::RankedVector<Index_, Index_> ref_ranked;
139 ref_ranked.reserve(NR);
140
141 std::vector<Float_> test_scaled(NR);
142 std::vector<Float_> ref_scaled(NR);
143 std::vector<Float_> all_correlations;
144
145 for (Index_ i = start, end = start + len; i < end; ++i) {
146 // Extracting only the markers of the best labels for this cell.
147 miniverse_tmp.clear();
148 for (size_t r = 0; r < nref; ++r) {
149 auto best = assigned[r][i];
150 const auto& markers = trained.markers[r][best];
151 miniverse_tmp.insert(markers.begin(), markers.end());
152 }
153
154 miniverse.clear();
155 miniverse.insert(miniverse.end(), miniverse_tmp.begin(), miniverse_tmp.end());
156 std::sort(miniverse.begin(), miniverse.end());
157
158 test_ranked_full.clear();
159 auto ptr = wrk->fetch(buffer.data());
160 for (auto u : miniverse) {
161 test_ranked_full.emplace_back(ptr[u], u);
162 }
163 std::sort(test_ranked_full.begin(), test_ranked_full.end());
164
165 // Scanning through each reference and computing the score for the best group.
166 Float_ best_score = -1000, next_best = -1000;
167 Index_ best_ref = 0;
168 bool direct_mapping_filled = false;
169
170 for (size_t r = 0; r < nref; ++r) {
171 // Further subsetting to the intersection of markers that are
172 // actual present in this particular reference.
173 const internal::RankRemapper<Index_>* mapping;
174 if (trained.check_availability[r]) {
175 const auto& cur_available = trained.available[r];
176 intersect_mapping.clear();
177 intersect_mapping.reserve(miniverse.size());
178 for (auto c : miniverse) {
179 if (cur_available.find(c) != cur_available.end()) {
180 intersect_mapping.add(c);
181 }
182 }
183 mapping = &intersect_mapping;
184
185 } else {
186 if (!direct_mapping_filled) {
187 direct_mapping.clear();
188 direct_mapping.reserve(miniverse.size());
189 for (auto c : miniverse) {
190 direct_mapping.add(c);
191 }
192 direct_mapping_filled = true;
193 }
194 mapping = &direct_mapping;
195 }
196
197 mapping->remap(test_ranked_full, test_ranked);
198 test_scaled.resize(test_ranked.size());
199 internal::scaled_ranks(test_ranked, test_scaled.data());
200
201 // Now actually calculating the score for the best group for
202 // this cell in this reference. This assumes that
203 // 'trained.ranked' already contains sorted pairs where the
204 // indices refer to the rows of the original data matrix.
205 auto best = assigned[r][i];
206 const auto& best_ranked = trained.ranked[r][best];
207 all_correlations.clear();
208 ref_scaled.resize(test_scaled.size());
209
210 for (size_t s = 0; s < best_ranked.size(); ++s) {
211 ref_ranked.clear();
212 mapping->remap(best_ranked[s], ref_ranked);
213 internal::scaled_ranks(ref_ranked, ref_scaled.data());
214 Float_ cor = internal::distance_to_correlation<Float_>(test_scaled, ref_scaled);
215 all_correlations.push_back(cor);
216 }
217
218 Float_ score = internal::correlations_to_scores(all_correlations, options.quantile);
219 if (buffers.scores[r]) {
220 buffers.scores[r][i] = score;
221 }
222 if (score > best_score) {
223 next_best = best_score;
224 best_score = score;
225 best_ref = r;
226 } else if (score > next_best) {
227 next_best = score;
228 }
229 }
230
231 if (buffers.best) {
232 buffers.best[i] = best_ref;
233 }
234 if (buffers.delta) {
235 if (nref > 1) {
236 buffers.delta[i] = best_score - next_best;
237 } else {
238 buffers.delta[i] = std::numeric_limits<Float_>::quiet_NaN();
239 }
240 }
241 }
242
243 }, test.ncol(), options.num_threads);
244}

300template<typename RefLabel_, typename Value_, typename Index_, typename Label_, typename Float_>
301ClassifyIntegratedResults<RefLabel_, Float_> classify_integrated(
302 const tatami::Matrix<Value_, Index_>& test,
303 const std::vector<const Label_*>& assigned,
304 const TrainedIntegrated<Index_>& trained,
305 const ClassifyIntegratedOptions<Float_>& options)
306{
307 ClassifyIntegratedResults<RefLabel_, Float_> results(test.ncol(), trained.num_references());
308 ClassifyIntegratedBuffers<RefLabel_, Float_> buffers;
309 buffers.best = results.best.data();
310 buffers.delta = results.delta.data();
311 buffers.scores.reserve(results.scores.size());
312 for (auto& s : results.scores) {
313 buffers.scores.emplace_back(s.data());
314 }
315 classify_integrated(test, assigned, trained, buffers, options);
316 return results;
317}
